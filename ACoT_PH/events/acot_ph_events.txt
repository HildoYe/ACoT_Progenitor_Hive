namespace = acot_ph

country_event = {
    id = acot_ph.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
		has_origin = origin_progenitor_hive
        OR = {
            last_increased_tech = tech_precursor_escort_delta
            last_increased_tech = tech_precursor_escort_alpha
            last_increased_tech = tech_precursor_escort_sigma
            last_increased_tech = tech_omega_escort
            last_increased_tech = tech_precursor_escort_cost
        }
    }

    immediate = {
        if = {
			limit = {
				last_increased_tech = tech_precursor_escort_cost
			}
			shipsize_offspring_precursor_escort_build_speed_mult = 0.25
			shipsize_offspring_precursor_escort_hull_add = 2000
		}
		else = {			
			shipsize_offspring_precursor_escort_hull_add = 750
		}
    }
}
